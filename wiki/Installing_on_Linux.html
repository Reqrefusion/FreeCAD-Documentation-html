import FreeCAD as App import Part import math

  1. =====================参数设置=====================

D_core_out = 325 #钻杆芯管外径 D_core_in = 293 #芯管内孔 pile_dia = 600 #桩径 blade_thick = 25 #叶片厚度 pitch = 800 #螺距 rod_total_length = 6000 flange_od = 380 #法兰外径 flange_h = 300 #法兰高度 blade_width = (pile_dia - D_core_out)/2

  1. =================================================

doc = App.newDocument("CompactionDrillRod")

  1. --------1 创建中心空心钻杆基体 --------

core_outer = Part.Cylinder(App.Vector(0,0,0), App.Vector(0,0,rod_total_length), D_core_out/2) core_inner = Part.Cylinder(App.Vector(0,0,0), App.Vector(0,0,rod_total_length), D_core_in/2) core_tube = core_outer.cut(core_inner)

  1. --------2 生成螺旋线路径(右旋螺旋) --------

helix_points = [] turn_num = (rod_total_length - flange_h)/pitch samples_per_loop = 120 for i in range(int(turn_num * samples_per_loop)+1): 

   t = i/samples_per_loop
   z = t * pitch
   ang = 2 * math.pi * t
   r = D_core_out/2
   x = r * math.cos(ang)
   y = r * math.sin(ang)
   helix_points.append(App.Vector(x,y,z))

helix = Part.BSplineCurve(helix_points)

  1. --------3 叶片截面草图：矩形(径向+厚度方向)扫描轮廓 --------
  2. 垂直于螺旋线局部坐标系，矩形截面：径向blade_width，厚度blade_thick

def create_blade_profile(): 

   pts = [
       App.Vector(D_core_out/2, -blade_thick/2, 0),
       App.Vector(D_core_out/2 + blade_width, -blade_thick/2, 0),
       App.Vector(D_core_out/2 + blade_width, blade_thick/2, 0),
       App.Vector(D_core_out/2, blade_thick/2, 0)
   ]
   poly = Part.Polygon(pts)
   return Part.Face(poly)

profile_face = create_blade_profile() blade_solid = Part.makeSweepSurface(helix, profile_face) blade_solid = Part.Solid(blade_solid)

  1. --------4 上端法兰 --------

flange_body = Part.Cylinder(App.Vector(0,0,rod_total_length - flange_h), 

                           App.Vector(0,0,flange_h), flange_od/2)
  1. 在法兰圆周阵列4个螺栓孔

bolt_dia = 30 bolt_r = 150 holes = [] for angle in [0, math.pi/2, math.pi, math.pi*3/2]: 

   bx = bolt_r * math.cos(angle)
   by = bolt_r * math.sin(angle)
   hole = Part.Cylinder(App.Vector(bx,by,rod_total_length-flange_h),
                        App.Vector(0,0,flange_h), bolt_dia/2)
   holes.append(hole)

flange_final = flange_body for h in holes: 

   flange_final = flange_final.cut(h)
  1. --------5 下端锥形钻尖简化模型 --------

cone_h = 800 cone_top_r = D_core_out/2 cone_bottom_r = 60 cone = Part.Cone(App.Vector(0,0,0), App.Vector(0,0,cone_h), cone_top_r, cone_bottom_r)

  1. --------组合全部零件 --------

whole = core_tube.fuse(blade_solid) whole = whole.fuse(flange_final) whole = whole.fuse(cone) whole = whole.removeSplitter()

Part.show(whole) doc.recompute() App.ActiveDocument.ActiveObject.ViewObject.Transparency = 0 print("模型生成完成：挤土钻杆三维模型")
