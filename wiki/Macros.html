import FreeCAD as App import Part

doc = App.newDocument("Tank_IPAL_Full")

  1. =========================
  2. PARAMETER (mm)
  3. =========================

D = 2400 R = D/2 H_tab = 2800 H_cone = 1800 t = 8 H_leg = 2200

  1. =========================
  2. TANGKI (OUTER)
  3. =========================

cyl = Part.makeCylinder(R, H_tab) cone = Part.makeCone(R, 0, H_cone) cyl.translate(App.Vector(0,0,H_cone)) outer = cyl.fuse(cone)

  1. =========================
  2. INNER (HOLLOW)
  3. =========================

Ri = R - t

cyl_i = Part.makeCylinder(Ri, H_tab - t) cone_i = Part.makeCone(Ri, 0, H_cone - t)

cyl_i.translate(App.Vector(0,0,H_cone + t)) cone_i.translate(App.Vector(0,0,t))

tank = outer.cut(cyl_i.fuse(cone_i))

  1. =========================
  2. OUTLET
  3. =========================

nozzle = Part.makeCylinder(150, 300) nozzle.translate(App.Vector(0,0,-300))

flange = Part.makeCylinder(220, 40) flange.translate(App.Vector(0,0,-40))

tank = tank.fuse(nozzle).fuse(flange)

  1. =========================
  2. KAKI (4 SUPPORT)
  3. =========================

leg_radius = 100 for x,y in [(1,1),(-1,1),(1,-1),(-1,-1)]: 

   leg = Part.makeCylinder(leg_radius, H_leg)
   leg.translate(App.Vector(x*(R+200), y*(R+200), -H_leg))
   tank = tank.fuse(leg)
  1. =========================
  2. RING PLATFORM
  3. =========================

ring_outer = Part.makeCylinder(R+300, 50) ring_inner = Part.makeCylinder(R+100, 50) ring = ring_outer.cut(ring_inner) ring.translate(App.Vector(0,0,H_cone + H_tab - 200))

tank = tank.fuse(ring)

  1. =========================
  2. LADDER
  3. =========================

ladder_h = H_tab + H_cone rail1 = Part.makeCylinder(20, ladder_h) rail2 = Part.makeCylinder(20, ladder_h)

rail1.translate(App.Vector(R+250,0,0)) rail2.translate(App.Vector(R+350,0,0))

tank = tank.fuse(rail1).fuse(rail2)

  1. anak tangga

for i in range(15): 

   step = Part.makeCylinder(10, 300)
   step.rotate(App.Vector(0,0,0), App.Vector(0,1,0), 90)
   step.translate(App.Vector(R+250,0,i*300))
   tank = tank.fuse(step)
  1. =========================
  2. MANHOLE + VENT
  3. =========================

manhole = Part.makeCylinder(250, 200) manhole.translate(App.Vector(0,0,H_cone + H_tab))

vent = Part.makeCylinder(50,150) vent.translate(App.Vector(400,0,H_cone + H_tab))

tank = tank.fuse(manhole).fuse(vent)

  1. =========================
  2. SHOW & EXPORT
  3. =========================

Part.show(tank) doc.recompute()

Part.export([tank], "Tank_IPAL_15m3_FULL.step") Part.export([tank], "Tank_IPAL_15m3_FULL.stl")

print("Model lengkap berhasil dibuat")
