import FreeCAD as App import Part import math

doc = App.newDocument("OMC_Diamond")

  1. Parameters

length = 6.0 width = 5.8 depth = 3.8

table_ratio = 0.50 culet_ratio = 0.05

crown_angle = math.radians(41) pavilion_angle = math.radians(42)

  1. Derived

table_size = length * table_ratio culet_size = length * culet_ratio

half_l = length / 2 half_w = width / 2

  1. Create girdle (simple cushion approximation)

girdle_pts = [ 

   App.Vector(-half_l, -half_w, 0),
   App.Vector( half_l, -half_w, 0),
   App.Vector( half_l,  half_w, 0),
   App.Vector(-half_l,  half_w, 0),

]

girdle_wire = Part.makePolygon(girdle_pts + [girdle_pts[0]]) girdle_face = Part.Face(girdle_wire)

  1. Crown height

crown_h = depth * 0.16 pavilion_h = depth * 0.43

  1. Table (top)

table_pts = [ 

   App.Vector(-table_size/2, -table_size/2, crown_h),
   App.Vector( table_size/2, -table_size/2, crown_h),
   App.Vector( table_size/2,  table_size/2, crown_h),
   App.Vector(-table_size/2,  table_size/2, crown_h),

]

table_wire = Part.makePolygon(table_pts + [table_pts[0]]) table_face = Part.Face(table_wire)

  1. Culet (bottom point)

culet = App.Vector(0, 0, -pavilion_h)

  1. Build crown (loft)

crown = Part.makeLoft([girdle_face, table_face], True)

  1. Build pavilion (loft to point)

pavilion_pts = [culet] pavilion_wire = Part.makePolygon([culet, culet]) # dummy pavilion = Part.makeLoft([girdle_face, Part.Vertex(culet).toShape()], True)

  1. Combine

diamond = crown.fuse(pavilion)

Part.show(diamond) doc.recompute()
