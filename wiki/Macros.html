import FreeCAD as App import Part import ImportGui import os

  1. Create a new document

doc = App.newDocument("Positioner_Scaled_Faceplate400")

  1. Photo aspect ratio ~16:9

ASPECT_W_OVER_H = 16.0 / 9.0

  1. Reference from you:

FACEPLATE_D_REF = 400.0 # mm

  1. In this model silhouette: faceplate_d = W * 0.16

W = FACEPLATE_D_REF / 0.16 H = W / ASPECT_W_OVER_H

S = 1.0 # global scaling

  1. Simplified proportions (edit anytime)

base_arm_len = W * 0.34 * S base_arm_w = W * 0.08 * S base_thk = H * 0.05 * S

column_d = W * 0.08 * S column_h = H * 0.62 * S

head_box_L = W * 0.16 * S head_box_W = W * 0.10 * S head_box_H = H * 0.12 * S

arm_len = W * 0.40 * S arm_d = W * 0.05 * S

faceplate_d = FACEPLATE_D_REF * S faceplate_thk = H * 0.04 * S

pad_L = W * 0.18 * S pad_W = W * 0.10 * S pad_H = base_thk * 0.70

  1. --- Build geometry ---

beam1 = Part.makeBox(base_arm_len * 2, base_arm_w, base_thk) beam2 = Part.makeBox(base_arm_w, base_arm_len * 2, base_thk) beam1.translate(App.Vector(-base_arm_len, -base_arm_w / 2, 0)) beam2.translate(App.Vector(-base_arm_w / 2, -base_arm_len, 0)) base = beam1.fuse(beam2)

column = Part.makeCylinder(column_d / 2, column_h) column.translate(App.Vector(0, 0, base_thk))

head = Part.makeBox(head_box_L, head_box_W, head_box_H) head.translate(App.Vector(-head_box_L * 0.35, -head_box_W / 2, base_thk + column_h - head_box_H * 0.65))

arm = Part.makeCylinder(arm_d / 2, arm_len) arm.rotate(App.Vector(0, 0, 0), App.Vector(0, 1, 0), 90) # point along X arm.translate(App.Vector(-arm_len - head_box_L * 0.25, 0, base_thk + column_h))

faceplate = Part.makeCylinder(faceplate_d / 2, faceplate_thk) faceplate.rotate(App.Vector(0, 0, 0), App.Vector(0, 1, 0), 90) faceplate.translate(App.Vector(-arm_len - faceplate_thk - head_box_L * 0.25, 0, base_thk + column_h))

pad = Part.makeBox(pad_L, pad_W, pad_H) pad.translate(App.Vector(base_arm_len * 0.25, base_arm_w / 2, 0))

model = base.fuse(column).fuse(head).fuse(arm).fuse(faceplate).fuse(pad)

obj = doc.addObject("Part::Feature", "Positioner") obj.Shape = model doc.recompute()

  1. --- Export STEP automatically to your Desktop ---

desktop = os.path.join(os.path.expanduser("~"), "Desktop") out_path = os.path.join(desktop, "welding_positioner_faceplate400.step")

ImportGui.export([obj], out_path) App.Console.PrintMessage(f"✅ STEP exported to: {out_path}\n")
