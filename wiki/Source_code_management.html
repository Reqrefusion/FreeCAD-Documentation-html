import FreeCAD as App import Part import ImportGui

  1. Create document

doc = App.newDocument("Meiso_E1607")

  1. Dimensions (mm)

L = 18.0 W = 16.0 H = 17.5

  1. Pin dimensions

pin_w = 0.6 pin_t = 0.6 pin_len = 3.5

  1. Main body

body = Part.makeBox(L, W, H)

  1. Pin locations

x_positions = [2.7, 5.4, 8.1, 10.8, 13.5]

pins = []

for x in x_positions: 

   # Bottom row
   pin1 = Part.makeBox(pin_w, pin_t, pin_len)
   pin1.translate(
       App.Vector(
           x - pin_w / 2,
           -pin_t / 2,
           -pin_len
       )
   )
   pins.append(pin1)

   # Top row
   pin2 = Part.makeBox(pin_w, pin_t, pin_len)
   pin2.translate(
       App.Vector(
           x - pin_w / 2,
           W - pin_t / 2,
           -pin_len
       )
   )
   pins.append(pin2)
  1. Fuse all shapes

shape = body

for pin in pins: 

   shape = shape.fuse(pin)
  1. Create feature

obj = doc.addObject("Part::Feature", "Meiso_E1607") obj.Shape = shape

doc.recompute()

  1. Export STEP

step_file = "Meiso_E1607.step" ImportGui.export([obj], step_file)

print(f"STEP file exported: {step_file}")
