Dieses Makro setzt die Position der Symbolleisten zurück. Das Makro in einem Arbeitsbereich ausführen, in der eine oder mehrere Symbolleisten fehlen.

import FreeCADGui as gui
from PySide import QtGui, QtCore # FreeCAD's special PySide!

mw = gui.getMainWindow()
tb = mw.findChildren(QtGui.QToolBar)
for i in tb:
    mw.addToolBar(QtCore.Qt.TopToolBarArea, i)
