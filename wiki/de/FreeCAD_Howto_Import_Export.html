// === PARAMETER ===

width = 100;

depth = 80;

height = 60;

wall = 3;

// Lüftungslöcher

hole_diameter = 4;

hole_spacing = 10;

// === GEHÄUSE ===

difference() {

    // Aussenbox

    cube([width, depth, height], false);

    // Innenraum aushöhlen

    translate([wall, wall, wall])

        cube([width - 2*wall, depth - 2*wall, height - wall], false);

    // Lüftungslöcher Seiten

    for (x = [10:hole_spacing:width-10]) {

        for (z = [10:hole_spacing:height-10]) {

            translate([x, 0, z])

                rotate([90,0,0])

                cylinder(h=depth, d=hole_diameter);

        }

    }

}

// === DECKEL ===

translate([0, depth + 10, 0])

difference() {

    cube([width, depth, wall]);

    // Lüftung im Deckel

    for (x = [10:hole_spacing:width-10]) {

        for (y = [10:hole_spacing:depth-10]) {

            translate([x, y, 0])

                cylinder(h=wall+1, d=hole_diameter);

        }

    }

}

// === HALTERUNGEN FÜR PLATINE ===

translate([20, 20, 5])

cylinder(h=5, d=5);

translate([80, 20, 5])

cylinder(h=5, d=5);

translate([20, 60, 5])

cylinder(h=5, d=5);

translate([80, 60, 5])

cylinder(h=5, d=5);
