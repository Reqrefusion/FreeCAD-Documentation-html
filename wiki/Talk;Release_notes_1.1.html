// Basic dimensions (you can edit these)

width = 120;

height = 100;

depth = 40;

thickness = 4;

// Main unit

module main_box() {

    difference() {

        cube([width, height, depth]);

        // Hollow inside

        translate([thickness, thickness, thickness])

            cube([width - 2*thickness, height - 2*thickness, depth - thickness]);

    }

}

// Shelf

module shelf(y_pos) {

    translate([thickness, y_pos, thickness])

        cube([width - 2*thickness, thickness, depth - thickness]);

}

// Drawer

module drawer() {

    translate([thickness, thickness, thickness])

        cube([width - 2*thickness, 20, depth - thickness]);

    // Knob

    translate([width/2, thickness + 10, depth + 2])

        sphere(2);

}

// Side extension (right box)

module side_box() {

    translate([width, 0, 0])

        difference() {

            cube([60, 60, depth]);

            translate([4, 4, 4])

                cube([52, 52, depth]);

            // Heart cut (approx)

            translate([30, 30, depth/2])

                scale([10,10,2])

                heart();

        }

}

// Simple heart shape

module heart() {

    union() {

        translate([-0.5,0,0]) circle(1);

        translate([0.5,0,0]) circle(1);

        translate([0,-0.7,0]) square([1.5,1.5], center=true);

    }

}

// Assembly

main_box();

shelf(50);

drawer();
