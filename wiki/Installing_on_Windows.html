import FreeCAD as App import Part

doc = App.newDocument("U_Model")

  1. 基本参数（根据图片反推）

tube_r = 8 # 管半径 height = 40 # 内高 width = 30 # 外宽 leg_len = 50 # 腿长

  1. 创建路径（U形）

p1 = App.Vector(0,0,0) p2 = App.Vector(0,0,leg_len) p3 = App.Vector(width,0,leg_len) p4 = App.Vector(width,0,0)

line1 = Part.makeLine(p1, p2) arc = Part.makeCircle(width/2, App.Vector(width/2,0,leg_len), App.Vector(0,1,0), 180, 0) line2 = Part.makeLine(p3, p4)

path = Part.Wire([line1, arc, line2])

  1. 截面（圆）

circle = Part.makeCircle(tube_r) profile = Part.Wire(circle)

  1. 扫描生成主体

pipe = Part.Wire(profile).makePipe(path)

  1. 底部加粗（模拟“鼓包”）

base1 = Part.makeSphere(12, App.Vector(0,0,0)) base2 = Part.makeSphere(12, App.Vector(width,0,0))

shape = pipe.fuse(base1).fuse(base2)

Part.show(shape)

  1. 导出STEP

Part.export([shape], "C:/Users/Public/u_shape_high_sim.stp")
