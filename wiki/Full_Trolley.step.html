  1. ===============================
  2. PARAMETERS (mm)
  3. ===============================

L = 800 W = 600 H = 900 PROFILE = 40

CASTER_DIA = 100 CASTER_H = 125

TOP_THK = 6 MONITOR_POST_H = 300

DRAWER_W = L - 2*PROFILE - 80 DRAWER_D = W - 2*PROFILE - 80 DRAWER_H = 120 DRAWER_Z = PROFILE + 460

def box(x,y,z,dx,dy,dz,name): 

   b = Part.makeBox(dx,dy,dz)
   b.translate(App.Vector(x,y,z))
   o = doc.addObject("Part::Feature", name)
   o.Shape = b
   return o

def caster(x,y): 

   c = Part.makeCylinder(CASTER_DIA/2,CASTER_H)
   c.translate(App.Vector(x,y,-CASTER_H))
   o = doc.addObject("Part::Feature","Caster")
   o.Shape = c

caster(PROFILE,PROFILE) caster(L-PROFILE,PROFILE) caster(PROFILE,W-PROFILE) caster(L-PROFILE,W-PROFILE)

doc.recompute()
