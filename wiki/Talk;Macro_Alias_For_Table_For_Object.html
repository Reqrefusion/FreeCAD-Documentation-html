def cell_to_rowcol(cell):

    """'AB12' -> (row=11, col=27)  — both zero-based."""

    m = re.match(r'^([A-Z]+)(\d+)$', cell.upper())

    if not m:

        raise ValueError(f"Invalid cell address: {cell!r}")

    col_str, row_str = m.group(1), m.group(2)

    col = 0

    for ch in col_str:

        col = col * 26 + (ord(ch) - ord('A') + 1)

    col -= 1   # make 0-based

    row = int(row_str) - 1

    return row, col

def rowcol_to_cell(row, col):

    """(row=11, col=27) -> 'AB12'  — both zero-based."""

    col_index = col

    col_str   = ''

    while True:

        col_str   = chr(col_index % 26 + ord('A')) + col_str

        col_index = col_index // 26 - 1

        if col_index < 0:

            break

    return col_str + str(row + 1)

def safe_get_contents(sheet, cell):

    """Return raw cell content as str, or '' — handles API name variation."""

    for method_name in ('getContents', 'getContent'):

        fn = getattr(sheet, method_name, None)

        if fn is not None:

            try:

                return str(fn(cell))

            except Exception:

                pass

    return ''

def traitementChaineSource(text):

    """Build a valid alias string from cell text.

    e.g. "Diametre du cylindre" -> "DiametreDuCylindre"

    """

    def replaceEquiv(ch):

        for pair in caracEquivalents:

            if pair[0] == ch:

                return pair[1]

        return ch

    def cleanWord(word):

        return ''.join(replaceEquiv(c) for c in word)

    result = ''

    first  = True

    for word in text.split(separateur):

        word = cleanWord(word)

        if first:

            result = word

            first  = False

        else:

            if majuscule:

                result += nouveauCaract + word[:1].upper() + word[1:]

            else:

                result += nouveauCaract + word

    if premierCaractereEnMinuscule:

        result = result[:1].lower() + result[1:]

    return result

def create_alias_for_table_cells_Selected():

    sheet     = get_active_sheet()

    if sheet is None:

        return

    sel_cells = get_selected_cells()

    if not sel_cells:

        return

    # Convert to (row, col) pairs and find the top-left origin

    rc_list = [cell_to_rowcol(c) for c in sel_cells]

    min_row  = min(r for r, c in rc_list)

    min_col  = min(c for r, c in rc_list)

    changed = 0

    for (row, col) in rc_list:

        # Skip header row and header column

        if row == min_row or col == min_col:

            continue

        cellName      = rowcol_to_cell(row, col)

        rowHeaderCell = rowcol_to_cell(row,     min_col)   # left edge

        colHeaderCell = rowcol_to_cell(min_row, col)       # top edge

        # Read each header independently so one empty header is allowed.

        # An empty/missing cell returns '' rather than causing a skip.

        def read_header(cell):

            try:

                val = sheet.get(cell)

                return str(val) if val is not None else ''

            except Exception:

                return ''

        textRow = read_header(rowHeaderCell)

        textCol = read_header(colHeaderCell)

        # Skip only if BOTH headers are empty (nothing to name the alias with)

        if not textRow and not textCol:

            continue

        # Build alias: if one side is empty, omit it and the separator

        if textRow and textCol:

            rawAlias = textRow + '_' + textCol

        elif textRow:

            rawAlias = textRow

        else:

            rawAlias = textCol

        textAlias = traitementChaineSource(rawAlias)

        if not textAlias:

            continue

        if addNumberIfOther:

            for i in range(1, 201):

                candidate = textAlias if i == 1 else f"{textAlias}_{i - 1}"

                try:

                    sheet.setAlias(cellName, candidate)

                    changed += 1

                    break

                except Exception:

                    continue

        else:

            try:

                sheet.setAlias(cellName, textAlias)

                changed += 1

            except Exception as e:

                App.Console.PrintWarning(

                    f"Alias_For_Table: Could not set alias '{textAlias}' "

                    f"on {cellName}: {e}\n"

                )

    if changed:

        App.ActiveDocument.recompute()

        App.Console.PrintMessage(

            f"Alias_For_Table: {changed} alias(es) created/updated.\n"

        )

    else:

        App.Console.PrintWarning(

            "Alias_For_Table: No aliases were created. "

            "Check that your selection includes both header row/column and data cells.\n"

        )

def set_constraint_expression(sketchLabel, constraint_name, expression):

    found    = False

    sketches = App.ActiveDocument.getObjectsByLabel(sketchLabel)

    if len(sketches) > 1:

        App.Console.PrintWarning(

            f"Alias_For_Table: Multiple sketches share label '{sketchLabel}'.\n"

        )

    for sketch in sketches:

        for i in range(sketch.ConstraintCount):

            if sketch.Constraints[i].Name == constraint_name:

                sketch.setExpression(f'Constraints[{i}]', expression)

                found = True

                break

    return found

def set_property_based_on_alias(sheet, cell):

    alias = sheet.getAlias(cell)

    if safe_get_contents(sheet, cell) == '':

        return

    if alias is None:

        App.Console.PrintWarning(

            f"Alias_For_Table: Cell {cell} has no alias – skipping.\n"

        )

        return

    parts = alias.rsplit('_', 1)

    if len(parts) != 2:

        App.Console.PrintWarning(

            f"Alias_For_Table: Alias '{alias}' is not 'ObjectLabel_Property' – skipping.\n"

        )

        return

    objLabel, property_keyword = parts

    matches = App.ActiveDocument.getObjectsByLabel(objLabel)

    if not matches:

        App.Console.PrintWarning(

            f"Alias_For_Table: No object with label '{objLabel}'.\n"

        )

        return

    obj           = matches[0]

    property_name = dico.get(property_keyword, property_keyword)

    expression    = f"<<{sheet.Label}>>.{alias}"

    if obj.TypeId == 'Sketcher::SketchObject':

        if set_constraint_expression(objLabel, property_keyword, expression):

            return

    top_level = property_name.split('.')[0]

    if hasattr(obj, top_level):

        try:

            obj.setExpression(property_name, expression)

        except Exception as e:

            App.Console.PrintError(

                f"Alias_For_Table: Cannot set expression on "

                f"'{objLabel}.{property_name}': {e}\n"

            )

    else:

        App.Console.PrintWarning(

            f"Alias_For_Table: Object '{objLabel}' has no property '{property_name}'.\n"

        )

def tableToObject():

    sheet     = get_active_sheet()

    if sheet is None:

        return

    sel_cells = get_selected_cells()

    if not sel_cells:

        return

    changed = 0

    for cell in sel_cells:

        if safe_get_contents(sheet, cell) != '':

            set_property_based_on_alias(sheet, cell)

            changed += 1

    if changed:

        App.ActiveDocument.recompute()

        App.Console.PrintMessage(

            f"Alias_For_Table: Expressions set for {changed} cell(s).\n"

        )
