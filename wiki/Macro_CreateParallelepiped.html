This is my first macro that makes a parallelepiped using height, length, width value

from PySide import QtGui
App.activeDocument().addObject('Part::Box','Box')
myNewHeight = QtGui.QInputDialog.getText(None, "Enter a value","Height")[0]
myNewWidth = QtGui.QInputDialog.getText(None, "Enter a value","Width")[0]
myNewLength = QtGui.QInputDialog.getText(None, "Enter a value","Length")[0]
Box = FreeCAD.ActiveDocument.getObject("Box")
myNewHeight = float(myNewHeight)
Box.Height = myNewHeight
myNewWidth = float(myNewWidth)
Box.Width = myNewWidth
myNewLength = float(myNewLength)
Box.Length = myNewLength
