import adsk.core, adsk.fusion, traceback

def run(context): 

   ui = None
   try:
       app = adsk.core.Application.get()
       ui = app.userInterface
       design = app.activeProduct
       
       # Crear nuevo componente
       rootComp = design.rootComponent
       newComp = rootComp.occurrences.addNewComponent(adsk.core.Matrix3D.create())
       comp = newComp.component
       
       # Sketch base (Vista Superior)
       sketches = comp.sketches
       xyPlane = comp.xYConstructionPlane
       sketch = sketches.add(xyPlane)
       
       # Dibujar base
       lines = sketch.sketchCurves.sketchLines
       arcs = sketch.sketchCurves.sketchArcs
       points = sketch.sketchPoints
       
       # Rectángulo base con redondeos
       p1 = points.add(adsk.core.Point3D.create(-50, -17.5, 0))
       p2 = points.add(adsk.core.Point3D.create(50, -17.5, 0))
       p3 = points.add(adsk.core.Point3D.create(50, 17.5, 0))
       p4 = points.add(adsk.core.Point3D.create(-50, 17.5, 0))
       
       # Agujeros base
       center1 = adsk.core.Point3D.create(-30, 0, 0)
       center2 = adsk.core.Point3D.create(30, 0, 0)
       sketch.sketchCurves.sketchCircles.addByCenterRadius(center1, 5)  # Ø10
       sketch.sketchCurves.sketchCircles.addByCenterRadius(center2, 5)
       
       # Extruir base
       extrudes = comp.features.extrudeFeatures
       prof = sketch.profiles.item(0)
       extInput = extrudes.createInput(prof, adsk.fusion.FeatureOperations.NewBodyFeatureOperation)
       distance = adsk.core.ValueInput.createByReal(10)  # espesor 10mm
       extInput.setDistanceExtent(False, distance)
       baseBody = extrudes.add(extInput)
       
       ui.messageBox('Modelo 3D base creado. Ahora crea la oreja curva manualmente o dime y te doy más código.')
       
   except:
       if ui:
           ui.messageBox('Error: ' + traceback.format_exc())
