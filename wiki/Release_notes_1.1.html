// ========================== // ROVER CHASSIS (30cm x 13cm) // ==========================

// PARAMETERS length = 300; width = 130; thickness = 4;

// Hole settings motor_offset = 15; hole_r = 2.5;

// Raspberry Pi holes pi_x = 58; pi_y = 49; pi_r = 1.5;

// Battery bat_l = 100; bat_w = 60;

// ========================== // BASE WITH CUTS // ========================== difference() { 

   // Base plate
   cube([length, width, thickness]);

   // MOTOR HOLES
   for (x = [motor_offset, length - motor_offset])
       for (y = [motor_offset, width - motor_offset])
           translate([x, y, -2])
               cylinder(h=thickness + 4, r=hole_r);

   // PI HOLES
   for (x = [-pi_x/2, pi_x/2])
       for (y = [-pi_y/2, pi_y/2])
           translate([length/2 + x, width/2 + y, -2])
               cylinder(h=thickness + 4, r=pi_r);

   // BATTERY CUTOUT
   translate([(length - bat_l)/2, (width - bat_w)/2, -2])
       cube([bat_l, bat_w, thickness + 4]);

   // WIRE SLOTS
   for (i = [40:60:length-40])
       translate([i, width/2 - 2.5, -2])
           cube([10, 5, thickness + 4]);

}

// ========================== // FRONT SENSOR MOUNT // ========================== translate([length/2 - 20, width, thickness]) cube([40, 10, 20]);

// ========================== // CAMERA HOLDER // ========================== translate([length/2 - 15, width + 12, thickness + 10]) cube([30, 5, 20]);

       cylinder(h=5, r=2);
