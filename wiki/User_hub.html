import cadquery as cq

# === BASE ===

# Base rectangular 100 x 60 x 5

pieza = cq.Workplane("XY").box(100, 60, 5)

# === CUERPO TRONCO PIRAMIDAL ===

# Se levanta desde la base con altura 50 mm

top = (

    cq.Workplane("XY")

    .workplane(offset=5)              # altura base

    .rect(60, 40)                     # base inferior

    .workplane(offset=50)             # altura total

    .rect(30, 20)                     # cara superior más pequeña

    .loft(combine=True)

)

pieza = pieza.union(top)

# (Opcional) Aquí podrías agregar los cortes según los detalles de tu plano

# === EXPORTAR A STEP ===

cq.exporters.export(pieza, "pieza_modelada.step")

# === EXPORTAR A IGES ===

cq.exporters.export(pieza, "pieza_modelada.iges")
